-- ========== ОБХОД АНТИЧИТОВ ==========
local AntiCheatBypass = {
    -- Отключение детекторов
    disableDetectors = function()
        for _, v in pairs(game:GetDescendants()) do
            if v.Name:lower():match("anticheat") or v.Name:lower():match("antihack") or v.Name:lower():match("detect") then
                pcall(function() v.Disabled = true end)
                pcall(function() v.Enabled = false end)
            end
            if v.ClassName == "RemoteEvent" or v.ClassName == "RemoteFunction" then
                if v.Name:lower():match("check") or v.Name:lower():match("report") or v.Name:lower():match("log") then
                    pcall(function() v:Destroy() end)
                end
            end
        end
    end,
    
    -- Скрытие скрипта
    hideScript = function()
        local scriptContext = game:GetService("ScriptContext")
        local old = scriptContext.ErrorMessage
        scriptContext.ErrorMessage = function(...) return end
        wait(0.1)
        scriptContext.ErrorMessage = old
    end,
    
    -- Подмена метатаблиц
    hookMetatables = function()
        local oldNamecall = nil
        oldNamecall = hookmetamethod(game, "__namecall", function(self, ...)
            local method = getnamecallmethod()
            if method == "FireServer" and tostring(self):match("Remote") then
                return nil
            end
            return oldNamecall(self, ...)
        end)
    end,
    
    -- Уничтожение камер слежки
    destroyCameras = function()
        for _, v in pairs(game.Workspace:GetDescendants()) do
            if v.ClassName == "Camera" and v.Name ~= "CurrentCamera" then
                pcall(function() v:Destroy() end)
            end
        end
    end,
    
    -- Полный обход
    fullBypass = function()
        pcall(function() AntiCheatBypass.disableDetectors() end)
        pcall(function() AntiCheatBypass.hideScript() end)
        pcall(function() AntiCheatBypass.hookMetatables() end)
        pcall(function() AntiCheatBypass.destroyCameras() end)
        print("[✔] Античит обойден")
    end
}

-- Запуск обхода античита
AntiCheatBypass.fullBypass()

local weaponGiven = false
local weaponType = nil
local weaponHandle = nil

-- ========== ФУНКЦИЯ ВЫДАЧИ ОРУЖИЯ (ОДИН РАЗ) ==========
function giveWeaponOnce(weaponName, weaponTool)
    if weaponGiven then
        print("[!] Оружие уже выдано! Перезапустите скрипт для повторной выдачи.")
        return false
    end
    
    weaponGiven = true
    weaponType = weaponName
    
    -- Очищаем инвентарь перед выдачей (опционально)
    -- for _, v in pairs(plr.Backpack:GetChildren()) do
    --     if v:IsA("Tool") then v:Destroy() end
    -- end
    
    weaponTool.Parent = plr.Backpack
    
    -- Скрываем оружие от детекции
    pcall(function()
        for _, v in pairs(weaponTool:GetDescendants()) do
            if v.ClassName == "RemoteEvent" or v.ClassName == "RemoteFunction" then
                v.Disabled = true
            end
        end
    end)
    
    print("[✔] " .. weaponName .. " выдан (только один раз)")
    return true
end

-- ========== ФУНКЦИЯ ДЛЯ ПМ (С ОДНОРАЗОВОЙ ВЫДАЧЕЙ) ==========
function getPMOnce()
    if weaponGiven then
        print("[!] ПМ уже выдан! Перезапустите скрипт.")
        return
    end
    
    -- Ищем ПМ в игровом мире
    for i, v in pairs(game.Players:GetDescendants()) do
        if v.Name == "ПМ" then
            weaponGiven = true
            weaponType = "ПМ"
            weaponHandle = v
            v.Parent = plr.Backpack
            
            -- Дополнительная защита от удаления
            pcall(function()
                local tool = v
                tool.AncestryChanged:Connect(function()
                    if tool.Parent ~= plr.Backpack and tool.Parent ~= plr.Character then
                        wait(0.1)
                        tool.Parent = plr.Backpack
                    end
                end)
            end)
            
            print("[✔] ПМ выдан (только один раз)")
            return
        end
    end
    print("[!] ПМ не найден в игре")
end

-- ========== ФУНКЦИЯ ДЛЯ БЕСКОНЕЧНЫХ ПАТРОНОВ ==========
function setInfiniteAmmo()
    if not weaponGiven then
        print("[!] Сначала получите оружие!")
        return
    end
    
    local weapon = nil
    for _, v in pairs(plr.Backpack:GetChildren()) do
        if v.Name == "ПМ" then
            weapon = v
            break
        end
    end
    for _, v in pairs(plr.Character:GetChildren()) do
        if v.Name == "ПМ" then
            weapon = v
            break
        end
    end
    
    if weapon and weapon:FindFirstChild("RepValues") and weapon.RepValues:FindFirstChild("Mag") then
        weapon.RepValues.Mag.Value = 99999999
        print("[✔] Бесконечные патроны активированы")
        
        -- Обновляем патроны каждую секунду (защита от сброса)
        spawn(function()
            while wait(1) do
                if weapon and weapon.Parent and weapon.RepValues and weapon.RepValues.Mag then
                    weapon.RepValues.Mag.Value = 99999999
                end
            end
        end)
    else
        print("[!] Не удалось найти магазин для патронов")
    end
end

-- ========== ЗАЩИТА ОТ КИКОВ ==========
local AntiKick = {
    preventKick = function()
        local oldKick = game.Players.LocalPlayer.Kick
        game.Players.LocalPlayer.Kick = function() end
        
        local oldDestroy = game.Destroy
        game.Destroy = function(self)
            if self == game then return end
            return oldDestroy(self)
        end
    end,
    
    reconnectOnKick = function()
        game:GetService("Players").LocalPlayer.OnTeleport:Connect(function()
            wait(2)
            game:GetService("TeleportService"):Teleport(game.PlaceId)
        end)
    end
}

AntiKick.preventKick()
AntiKick.reconnectOnKick()

    print("[✔] Персонаж возрожден, оружие восстановлено")
