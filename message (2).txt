-- ========== ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ ==========
local weaponGiven = false
local currentWeapon = nil
local isProcessing = false
local antiCheatKilled = false

-- ========== 1. ПОЛНОЕ УНИЧТОЖЕНИЕ АНТИЧИТА ==========
local function fullAntiCheatKill()
    if antiCheatKilled then return end
    
    -- Уничтожение скриптов античита
    local scriptsToDestroy = {
        "AntiCheat", "AntiHack", "AntiExploit", "Security", "Guard",
        "Protection", "Monitor", "Logger", "Detector", "Admin", "Moderation"
    }
    
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Script") or v:IsA("LocalScript") or v:IsA("ModuleScript") then
            for _, name in pairs(scriptsToDestroy) do
                if v.Name:lower():find(name:lower()) then
                    pcall(function() 
                        v.Disabled = true
                        v:Destroy()
                    end)
                end
            end
        end
        
        -- Уничтожение удаленных событий
        if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
            local vName = v.Name:lower()
            if vName:find("report") or vName:find("log") or vName:find("cheat") or
               vName:find("hack") or vName:find("detect") or vName:find("ban") or
               vName:find("kick") or vName:find("ac_") or vName:find("anticheat") then
                pcall(function() v:Destroy() end)
            end
        end
    end
    
    antiCheatKilled = true
    print("[✔] Античит уничтожен")
end

-- ========== 2. БЛОКИРОВКА ЗВУКОВ (СКРИМЕРОВ) ==========
local function blockAllAudio()
    -- Отключение всех существующих звуков
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Sound") then
            pcall(function()
                v.Volume = 0
                v.Playing = false
                v:Stop()
            end)
        end
    end
    
    -- Перехват новых звуков
    local oldSound = Instance.new("Sound").Play
    hookfunction(oldSound, function(sound)
        local soundName = sound.Name:lower()
        local blockWords = {"scream", "scary", "jumpscare", "alert", "alarm", "warning", "ban", "kick"}
        for _, word in pairs(blockWords) do
            if soundName:find(word) then
                return nil
            end
        end
        return oldSound(sound)
    end)
    
    print("[✔] Звуки отключены")
end

-- ========== 3. ЗАЩИТА ОТ КИКА И БАНА ==========
local function antiKickBan()
    -- Блокировка кика
    pcall(function()
        game.Players.LocalPlayer.Kick = function() end
    end)
    
    -- Блокировка телепортации в бан
    pcall(function()
        local teleportService = game:GetService("TeleportService")
        local oldTeleport = teleportService.Teleport
        teleportService.Teleport = function(placeId, ...)
            if placeId ~= game.PlaceId then
                return nil
            end
            return oldTeleport(placeId, ...)
        end
    end)
    
    -- Блокировка уничтожения игры
    pcall(function()
        local oldDestroy = game.Destroy
        game.Destroy = function() end
    end)
    
    print("[✔] Защита от кика/бана активирована")
end

-- ========== 4. ПЕРЕХВАТ ЛОГОВ ==========
local function interceptLogs()
    pcall(function()
        local oldPrint = print
        print = function(...) 
            local args = {...}
            local argsStr = tostring(args)
            if argsStr:lower():find("cheat") or argsStr:lower():find("hack") then
                return nil
            end
            return oldPrint(...)
        end
    end)
    print("[✔] Логи перехвачены")
end

-- ========== 5. БЕЗОПАСНАЯ ВЫДАЧА ПМ (БЕЗ ДЮПА) ==========
local function givePMSafe()
    -- Проверка на повторную выдачу
    if weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ уже выдан! Перезапустите скрипт.",
            Duration = 3
        })
        return false
    end
    
    -- Защита от одновременного запуска
    if isProcessing then
        return false
    end
    isProcessing = true
    
    -- Запускаем защиту перед выдачей
    pcall(function()
        fullAntiCheatKill()
        blockAllAudio()
        antiKickBan()
        interceptLogs()
    end)
    
    -- Поиск ПМ в игровом мире
    local weapon = nil
    for _ = 1, 10 do
        for _, v in pairs(game:GetDescendants()) do
            if v.Name == "ПМ" and v:IsA("Tool") then
                if v.Parent ~= plr.Backpack and v.Parent ~= plr.Character then
                    weapon = v
                    break
                end
            end
        end
        if weapon then break end
        wait(0.3)
    end
    
    if not weapon then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ не найден в игре",
            Duration = 3
        })
        isProcessing = false
        return false
    end
    
    -- УДАЛЕНИЕ ВСЕХ КОПИЙ ПМ (ПРЕДОТВРАЩЕНИЕ ДЮПА)
    local copiesToDelete = {}
    for _, tool in pairs(plr.Backpack:GetChildren()) do
        if tool.Name == "ПМ" and tool ~= weapon then
            table.insert(copiesToDelete, tool)
        end
    end
    for _, tool in pairs(plr.Character:GetChildren()) do
        if tool.Name == "ПМ" and tool ~= weapon then
            table.insert(copiesToDelete, tool)
        end
    end
    for _, tool in pairs(copiesToDelete) do
        pcall(function() tool:Destroy() end)
    end
    
    -- Отключение скриптов внутри оружия
    for _, script in pairs(weapon:GetDescendants()) do
        if script:IsA("Script") or script:IsA("LocalScript") then
            pcall(function() script.Disabled = true end)
        end
    end
    
    -- Выдача оружия
    pcall(function() weapon.Parent = plr.Backpack end)
    weaponGiven = true
    currentWeapon = weapon
    
    -- Мониторинг для предотвращения дюпа
    spawn(function()
        while weaponGiven and currentWeapon and currentWeapon.Parent do
            wait(0.5)
            -- Удаляем случайно появившиеся копии
            for _, tool in pairs(plr.Backpack:GetChildren()) do
                if tool.Name == "ПМ" and tool ~= currentWeapon then
                    pcall(function() tool:Destroy() end)
                end
            end
            for _, tool in pairs(plr.Character:GetChildren()) do
                if tool.Name == "ПМ" and tool ~= currentWeapon then
                    pcall(function() tool:Destroy() end)
                end
            end
        end
    end)
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "Успех",
        Text = "ПМ выдан! Дюп невозможен.",
        Duration = 3
    })
    
    isProcessing = false
    return true
end

-- ========== 6. БЕСКОНЕЧНЫЕ ПАТРОНЫ ==========
local function infiniteAmmo()
    if not weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Сначала получите ПМ!",
            Duration = 3
        })
        return
    end
    
    local weapon = nil
    for _, tool in pairs(plr.Backpack:GetChildren()) do
        if tool.Name == "ПМ" then
            weapon = tool
            break
        end
    end
    if not weapon then
        for _, tool in pairs(plr.Character:GetChildren()) do
            if tool.Name == "ПМ" then
                weapon = tool
                break
            end
        end
    end
    
    if weapon and weapon:FindFirstChild("RepValues") and weapon.RepValues:FindFirstChild("Mag") then
        local mag = weapon.RepValues.Mag
        mag.Value = 99999999
        
        -- Постоянное обновление
        spawn(function()
            while weaponGiven and weapon and weapon.Parent do
                wait(0.3)
                pcall(function()
                    if weapon.RepValues and weapon.RepValues.Mag then
                        if weapon.RepValues.Mag.Value < 99999999 then
                            weapon.RepValues.Mag.Value = 99999999
                        end
                    end
                end)
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "Бесконечные патроны активированы",
            Duration = 2
        })
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Не удалось найти магазин",
            Duration = 3
        })
    end
end

-- ========== 8. ДЮП ФЕЙРВЕРКОВ ==========
local function dupeFireworks()
    pcall(function()
        for _, v in pairs(game.Workspace:GetDescendants()) do
            if v:IsA("ProximityPrompt") and v.Parent and v.Parent.Name == "FireworkGiver" then
                v.Enabled = true
                v.HoldDuration = 0
                v.RequiresLineOfSight = false
                v.MaxActivationDistance = 50
            end
        end
        if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
            plr.Character.HumanoidRootPart.CFrame = CFrame.new(-2406.3667, 7.48527145, 39.0829697)
        end
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "Дюп фейрверков активирован",
            Duration = 2
        })
    end)
end
