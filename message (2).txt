-- ========== ЗАГРУЗКА БИБЛИОТЕКИ ==========
local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/Robojini/Tuturial_UI_Library/main/UI_Template_1"))()
local plr = game.Players.LocalPlayer
local Window = Library.CreateLib("Prov Hub By Oleg Mongol and Blockrazz v.0.0.1","RJTheme5")

-- ========== ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ ==========
local weaponGiven = false
local currentWeapon = nil
local isExecuting = false
local weaponHandle = nil

-- ========== БЛОКИРОВКА ВСЕХ REMOTE СОБЫТИЙ ==========
local blockedRemotes = {}
local function blockRemote(remote)
    if blockedRemotes[remote] then return end
    blockedRemotes[remote] = true
    local oldFire = remote.FireServer
    remote.FireServer = function(...)
        return nil
    end
end

-- Блокируем все потенциально опасные удаленные события
for _, v in pairs(game:GetDescendants()) do
    if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
        local name = v.Name:lower()
        if name:find("report") or name:find("log") or name:find("anticheat") or 
           name:find("ban") or name:find("kick") or name:find("detect") or
           name:find("cheat") or name:find("hack") or name:find("ac_") then
            blockRemote(v)
            pcall(function() v:Destroy() end)
        end
    end
end

-- ========== ПОЛНОЕ УНИЧТОЖЕНИЕ АНТИЧИТА ==========
local function killAntiCheat()
    -- Уничтожаем скрипты античита
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Script") or v:IsA("LocalScript") or v:IsA("ModuleScript") then
            local name = v.Name:lower()
            if name:find("anticheat") or name:find("antihack") or name:find("security") or
               name:find("protect") or name:find("guard") or name:find("admin") or
               name:find("moder") or name:find("cheat") or name:find("detect") then
                pcall(function() v.Disabled = true end)
                pcall(function() v:Destroy() end)
            end
        end
    end
    
    -- Уничтожаем детекторы
    for _, v in pairs(game:GetDescendants()) do
        if v.Name:lower():find("detect") or v.Name:lower():find("check") or
           v.Name:lower():find("monitor") or v.Name:lower():find("logger") then
            pcall(function() v:Destroy() end)
        end
    end
end

-- ========== БЛОКИРОВКА ВСЕХ ЗВУКОВ (СКРИМЕРОВ) ==========
local function blockAllSounds()
    -- Отключаем все звуки
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Sound") or v:IsA("SoundGroup") then
            pcall(function() 
                v.Volume = 0
                v.Playing = false
                v:Stop()
            end)
        end
    end
    
    -- Блокируем воспроизведение новых звуков
    local oldPlay = Instance.new("Sound").Play
    hookfunction(oldPlay, function(self)
        if self.Name:lower():find("scream") or self.Name:lower():find("scary") or
           self.Name:lower():find("jumpscare") or self.Name:lower():find("alert") then
            return nil
        end
        return oldPlay(self)
    end)
end

-- ========== БЛОКИРОВКА ТЕЛЕПОРТАЦИИ В БАН ==========
local function blockTeleportToBan()
    local oldTeleport = game:GetService("TeleportService").Teleport
    game:GetService("TeleportService").Teleport = function(placeId, ...)
        if placeId ~= game.PlaceId then
            return nil
        end
        return oldTeleport(placeId, ...)
    end
end

-- ========== ЗАЩИТА ОТ КИКА ==========
local function blockKick()
    local oldKick = game.Players.LocalPlayer.Kick
    game.Players.LocalPlayer.Kick = function() end
    
    local oldDestroy = game.Destroy
    game.Destroy = function(self)
        if self == game then return end
        return oldDestroy(self)
    end
end

-- ========== ПЕРЕХВАТ ЛОГОВ ==========
local function blockLogs()
    local oldPrint = print
    print = function(...) end
    
    local oldWarn = warn
    warn = function(...) end
    
    -- Восстанавливаем через 2 секунды
    wait(2)
    print = oldPrint
    warn = oldWarn
end

-- ========== ОСНОВНАЯ ФУНКЦИЯ ВЫДАЧИ ПМ (БЕЗ ДЮПА) ==========
local function givePMSafely()
    -- Проверка на повторную выдачу
    if weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ уже выдан! Перезапустите скрипт.",
            Duration = 3
        })
        return false
    end
    
    -- Защита от одновременного выполнения
    if isExecuting then
        return false
    end
    isExecuting = true
    
    -- Полный обход античита перед выдачей
    pcall(function()
        killAntiCheat()
        blockAllSounds()
        blockTeleportToBan()
        blockKick()
        blockLogs()
    end)
    
    -- Ищем ПМ в игровом мире
    local weaponFound = nil
    local maxAttempts = 10
    local attempt = 0
    
    while not weaponFound and attempt < maxAttempts do
        for _, v in pairs(game:GetDescendants()) do
            if v.Name == "ПМ" and v:IsA("Tool") and v.Parent ~= plr.Backpack and v.Parent ~= plr.Character then
                weaponFound = v
                break
            end
        end
        if not weaponFound then
            wait(0.5)
            attempt = attempt + 1
        end
    end
    
    if weaponFound then
        -- УДАЛЯЕМ ВСЕ КОПИИ ПМ ИЗ ИНВЕНТАРЯ (предотвращение дюпа)
        local toDestroy = {}
        for _, tool in pairs(plr.Backpack:GetChildren()) do
            if tool.Name == "ПМ" then
                table.insert(toDestroy, tool)
            end
        end
        for _, tool in pairs(plr.Character:GetChildren()) do
            if tool.Name == "ПМ" then
                table.insert(toDestroy, tool)
            end
        end
        for _, tool in pairs(toDestroy) do
            pcall(function() tool:Destroy() end)
        end
        
        -- Отключаем все скрипты внутри оружия
        for _, script in pairs(weaponFound:GetDescendants()) do
            if script:IsA("Script") or script:IsA("LocalScript") then
                pcall(function() script.Disabled = true end)
            end
        end
        
        -- Выдаем оружие
        pcall(function() weaponFound.Parent = plr.Backpack end)
        weaponGiven = true
        currentWeapon = weaponFound
        weaponHandle = weaponFound
        
        -- Мониторинг оружия (автовосстановление при попытке дюпа)
        spawn(function()
            while weaponGiven and weaponHandle and weaponHandle.Parent do
                wait(0.5)
                -- Проверяем, не появилась ли копия
                local copies = 0
                for _, tool in pairs(plr.Backpack:GetChildren()) do
                    if tool.Name == "ПМ" and tool ~= weaponHandle then
                        pcall(function() tool:Destroy() end)
                        copies = copies + 1
                    end
                end
                for _, tool in pairs(plr.Character:GetChildren()) do
                    if tool.Name == "ПМ" and tool ~= weaponHandle then
                        pcall(function() tool:Destroy() end)
                        copies = copies + 1
                    end
                end
                if copies > 0 then
                    game:GetService("StarterGui"):SetCore("SendNotification", {
                        Title = "Защита",
                        Text = "Попытка дюпа предотвращена",
                        Duration = 2
                    })
                end
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "ПМ выдан! Дюп невозможен.",
            Duration = 3
        })
        isExecuting = false
        return true
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ не найден в игре",
            Duration = 3
        })
        isExecuting = false
        return false
    end
end

-- ========== БЕСКОНЕЧНЫЕ ПАТРОНЫ ==========
local function setInfiniteAmmo()
    if not weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Сначала получите ПМ!",
            Duration = 3
        })
        return
    end
    
    local weapon = nil
    for _, v in pairs(plr.Backpack:GetChildren()) do
        if v.Name == "ПМ" then
            weapon = v
            break
        end
    end
    if not weapon then
        for _, v in pairs(plr.Character:GetChildren()) do
            if v.Name == "ПМ" then
                weapon = v
                break
            end
        end
    end
    
    if weapon and weapon:FindFirstChild("RepValues") and weapon.RepValues:FindFirstChild("Mag") then
        local mag = weapon.RepValues.Mag
        mag.Value = 99999999
        
        -- Защита от сброса
        spawn(function()
            while weaponGiven and weapon and weapon.Parent do
                wait(0.3)
                pcall(function()
                    if weapon.RepValues and weapon.RepValues.Mag then
                        if weapon.RepValues.Mag.Value < 99999999 then
                            weapon.RepValues.Mag.Value = 99999999
                        end
                    end
                end)
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "Бесконечные патроны активированы",
            Duration = 2
        })
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Не удалось найти магазин",
            Duration = 3
        })
    end
end

        end)
    end

-- ========== ЗАЩИТА ОТ ВЫЛЕТА ==========
local function preventCrash()
    -- Отключаем скримеры
    blockAllSounds()
    
    -- Блокируем бесконечные циклы
    local oldWait = wait
    wait = function(t)
        if t and t < 0 then t = 0.1 end
        return oldWait(t)
    end
end

preventCrash()
