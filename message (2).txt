-- ========== ЗАГРУЗКА БИБЛИОТЕКИ ==========
local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/OlegMongolRoblox/Hacks/refs/heads/main/1"))()

-- ========== ПЕРЕМЕННЫЕ БЕЗОПАСНОСТИ ==========
local weaponGiven = false
local shieldGiven = false
local antiCheatDestroyed = false
local soundBlocked = false

-- ========== 1. ПОЛНОЕ УНИЧТОЖЕНИЕ АНТИЧИТА ==========
local function destroyAntiCheat()
    if antiCheatDestroyed then return true end
    
    local success = true
    
    -- Уничтожение скриптов античита
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Script") or v:IsA("LocalScript") or v:IsA("ModuleScript") then
            local name = v.Name:lower()
            local badNames = {"anticheat", "antihack", "security", "protect", "guard", 
                              "admin", "moder", "cheat", "detect", "monitor", "logger", 
                              "report", "ban", "kick", "ac_", "anti", "hack"}
            for _, bad in pairs(badNames) do
                if name:find(bad) then
                    pcall(function() v.Disabled = true end)
                    pcall(function() v:Destroy() end)
                    break
                end
            end
        end
        
        -- Уничтожение удаленных событий
        if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
            local name = v.Name:lower()
            local badRemote = {"report", "log", "cheat", "hack", "detect", "ban", "kick", "anticheat"}
            for _, bad in pairs(badRemote) do
                if name:find(bad) then
                    pcall(function() v:Destroy() end)
                    break
                end
            end
        end
    end
    
    antiCheatDestroyed = true
    return success
end

-- ========== 2. БЛОКИРОВКА ЗВУКОВ (СКРИМЕРОВ) ==========
local function blockAllSounds()
    if soundBlocked then return end
    
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Sound") then
            pcall(function()
                v.Volume = 0
                v.Playing = false
                v:Stop()
            end)
        end
    end
    
    soundBlocked = true
end

-- ========== 3. ЗАЩИТА ОТ КИКА И БАНА ==========
local function protectFromKick()
    pcall(function()
        game.Players.LocalPlayer.Kick = function() end
    end)
    
    pcall(function()
        local ts = game:GetService("TeleportService")
        local oldTeleport = ts.Teleport
        ts.Teleport = function(placeId, ...)
            if placeId ~= game.PlaceId then
                return nil
            end
            return oldTeleport(placeId, ...)
        end
    end)
end

-- ========== 4. ПЕРЕХВАТ ЛОГОВ ==========
local function interceptLogs()
    pcall(function()
        local oldPrint = print
        print = function(...)
            local args = {...}
            local str = tostring(args)
            if str:lower():find("cheat") or str:lower():find("hack") or 
               str:lower():find("exploit") or str:lower():find("inject") then
                return nil
            end
            return oldPrint(...)
        end
    end)
end

-- ========== 5. БЕЗОПАСНАЯ ВЫДАЧА ПМ (БЕЗ ДЮПА) ==========
local function safeGivePM()
    if weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Безопасность",
            Text = "ПМ уже выдан! Защита активна.",
            Duration = 2
        })
        return
    end
    
    -- Удаляем все копии ПМ перед выдачей
    for _, tool in pairs(plr.Backpack:GetChildren()) do
        if tool.Name == "ПМ" then
            tool:Destroy()
        end
    end
    for _, tool in pairs(plr.Character:GetChildren()) do
        if tool.Name == "ПМ" then
            tool:Destroy()
        end
    end
    
    -- Поиск ПМ
    local weapon = nil
    for _, v in pairs(game:GetDescendants()) do
        if v.Name == "ПМ" and v:IsA("Tool") then
            if v.Parent ~= plr.Backpack and v.Parent ~= plr.Character then
                weapon = v
                break
            end
        end
    end
    
    if weapon then
        weapon.Parent = plr.Backpack
        weaponGiven = true
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Безопасно",
            Text = "ПМ выдан. Дюп невозможен.",
            Duration = 2
        })
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ не найден",
            Duration = 2
        })
    end
end

-- ========== 7. БЕСКОНЕЧНЫЕ ПАТРОНЫ (БЕЗОПАСНЫЕ) ==========
local function safeInfiniteAmmo()
    if not weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Сначала получите ПМ",
            Duration = 2
        })
        return
    end
    
    local weapon = nil
    for _, tool in pairs(plr.Backpack:GetChildren()) do
        if tool.Name == "ПМ" then weapon = tool break end
    end
    if not weapon then
        for _, tool in pairs(plr.Character:GetChildren()) do
            if tool.Name == "ПМ" then weapon = tool break end
        end
    end
    
    if weapon and weapon:FindFirstChild("RepValues") and weapon.RepValues:FindFirstChild("Mag") then
        weapon.RepValues.Mag.Value = 99999999
        
        -- Цикл обновления для защиты от сброса
        spawn(function()
            while weaponGiven and weapon and weapon.Parent do
                wait(0.5)
                pcall(function()
                    if weapon.RepValues and weapon.RepValues.Mag then
                        if weapon.RepValues.Mag.Value < 99999999 then
                            weapon.RepValues.Mag.Value = 99999999
                        end
                    end
                end)
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Безопасно",
            Text = "Бесконечные патроны активированы",
            Duration = 2
        })
    end
end

        })
    end)
end
