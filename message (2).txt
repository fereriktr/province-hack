-- ========== ПОЛНЫЙ ОБХОД АНТИЧИТА ==========
local AntiCheat = {
    -- 1. Отключение всех удаленных событий античита
    killRemotes = function()
        local remotesToKill = {
            "AntiCheat", "AntiCheatLog", "ReportPlayer", "LogData", 
            "CheatDetected", "BanPlayer", "KickPlayer", "SendLogs",
            "PlayerReport", "AC_Check", "AntiExploit"
        }
        
        for _, v in pairs(game:GetDescendants()) do
            if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
                for _, name in pairs(remotesToKill) do
                    if v.Name:lower():find(name:lower()) then
                        pcall(function() v:Destroy() end)
                    end
                end
            end
        end
    end,
    
    -- 2. Уничтожение скриптов античита
    killScripts = function()
        local scriptsToKill = {
            "AntiCheat", "AntiExploit", "AC", "AntiHack", "Security",
            "PlayerMonitor", "GameGuard", "Protection", "Admin"
        }
        
        for _, v in pairs(game:GetDescendants()) do
            if v:IsA("Script") or v:IsA("LocalScript") or v:IsA("ModuleScript") then
                for _, name in pairs(scriptsToKill) do
                    if v.Name:lower():find(name:lower()) then
                        pcall(function() v.Disabled = true end)
                        pcall(function() v:Destroy() end)
                    end
                end
            end
        end
    end,
    
    -- 3. Перехват и блокировка отправки логов
    blockLogs = function()
        local oldFireServer = hookfunction(Instance.new("RemoteEvent").FireServer, function(self, ...)
            local args = {...}
            local argsStr = tostring(args)
            
            -- Блокируем любые подозрительные отправки
            local blockedKeywords = {"log", "report", "cheat", "hack", "ban", "kick", "detect", "ac_"}
            for _, keyword in pairs(blockedKeywords) do
                if argsStr:lower():find(keyword) then
                    return nil
                end
            end
            
            -- Блокируем отправку если имя remote содержит античит
            if self.Name:lower():find("anticheat") or self.Name:lower():find("report") then
                return nil
            end
            
            return oldFireServer(self, ...)
        end)
    end,
    
    -- 4. Скрытие факта выполнения скрипта
    hideExecution = function()
        -- Скрываем скрипт из стектрейса
        local oldError = error
        error = function(...) return nil end
        
        -- Отключаем вывод в консоль
        local oldPrint = print
        print = function(...) end
        
        -- Восстанавливаем через 1 секунду
        wait(1)
        print = oldPrint
        error = oldError
    end,
    
    -- 5. Уничтожение детекторов
    killDetectors = function()
        for _, v in pairs(game:GetDescendants()) do
            -- Отключаем все ProximityPrompt детекторы
            if v:IsA("ProximityPrompt") then
                v.Enabled = false
            end
            
            -- Уничтожаем кастомные детекторы
            if v.Name:lower():find("detect") or v.Name:lower():find("check") then
                pcall(function() v:Destroy() end)
            end
        end
    end,
    
    -- 6. Обход через метатаблицы
    hookGame = function()
        local mt = getrawmetatable(game)
        local old = mt.__namecall
        setreadonly(mt, false)
        
        mt.__namecall = newcclosure(function(self, ...)
            local method = getnamecallmethod()
            local args = {...}
            
            -- Блокируем любые проверки
            if method == "FireServer" then
                local selfStr = tostring(self)
                local argsStr = tostring(args)
                
                if selfStr:lower():find("anticheat") or argsStr:lower():find("detect") then
                    return nil
                end
            end
            
            -- Блокируем кик
            if method == "Kick" then
                return nil
            end
            
            return old(self, ...)
        end)
        
        setreadonly(mt, true)
    end,
    
    -- 7. Полный запуск обхода
    fullBypass = function()
        print("[*] Обход античита...")
        
        pcall(function() AntiCheat.killRemotes() end)
        pcall(function() AntiCheat.killScripts() end)
        pcall(function() AntiCheat.blockLogs() end)
        pcall(function() AntiCheat.killDetectors() end)
        pcall(function() AntiCheat.hookGame() end)
        pcall(function() AntiCheat.hideExecution() end)
        
        print("[✔] Античит полностью обойден")
    end
}

-- Запуск обхода
AntiCheat.fullBypass()

-- ========== ПЕРЕМЕННЫЕ ДЛЯ КОНТРОЛЯ ВЫДАЧИ ==========
local weaponGiven = false
local currentWeapon = nil

-- ========== ФУНКЦИЯ БЕЗОПАСНОЙ ВЫДАЧИ ОРУЖИЯ (НЕТ ДЮПА) ==========
function giveWeaponSafely()
    if weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Оружие уже выдано! Перезапустите скрипт.",
            Duration = 3
        })
        return false
    end
    
    -- Ищем оружие в игровом мире
    local weaponFound = nil
    for _, v in pairs(game.Players:GetDescendants()) do
        if v.Name == "ПМ" and v:IsA("Tool") then
            weaponFound = v
            break
        end
    end
    
    if not weaponFound then
        for _, v in pairs(game.Workspace:GetDescendants()) do
            if v.Name == "ПМ" and v:IsA("Tool") then
                weaponFound = v
                break
            end
        end
    end
    
    if weaponFound then
        -- Удаляем все копии оружия из инвентаря
        for _, tool in pairs(plr.Backpack:GetChildren()) do
            if tool.Name == "ПМ" then
                tool:Destroy()
            end
        end
        for _, tool in pairs(plr.Character:GetChildren()) do
            if tool.Name == "ПМ" then
                tool:Destroy()
            end
        end
        
        -- Выдаем оружие
        weaponFound.Parent = plr.Backpack
        weaponGiven = true
        currentWeapon = weaponFound
        
        -- Добавляем защиту от удаления
        local tool = weaponFound
        local connection
        connection = tool.AncestryChanged:Connect(function()
            if tool.Parent ~= plr.Backpack and tool.Parent ~= plr.Character then
                wait(0.1)
                if not tool.Parent then
                    pcall(function() tool.Parent = plr.Backpack end)
                end
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "ПМ выдан!",
            Duration = 2
        })
        return true
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ не найден в игре",
            Duration = 3
        })
        return false
    end
end

-- ========== ФУНКЦИЯ БЕСКОНЕЧНЫХ ПАТРОНОВ ==========
function setInfiniteAmmo()
    if not weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Сначала получите оружие!",
            Duration = 3
        })
        return
    end
    
    local weapon = nil
    for _, v in pairs(plr.Backpack:GetChildren()) do
        if v.Name == "ПМ" then
            weapon = v
            break
        end
    end
    for _, v in pairs(plr.Character:GetChildren()) do
        if v.Name == "ПМ" then
            weapon = v
            break
        end
    end
    
    if weapon and weapon:FindFirstChild("RepValues") and weapon.RepValues:FindFirstChild("Mag") then
        -- Устанавливаем бесконечные патроны
        local mag = weapon.RepValues.Mag
        mag.Value = 999999
        
        -- Защита от сброса
        spawn(function()
            while wait(0.5) do
                if mag and mag.Parent then
                    if mag.Value < 999999 then
                        mag.Value = 999999
                    end
                end
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "Бесконечные патроны активированы",
            Duration = 2
        })
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Не удалось найти магазин",
            Duration = 3
        })
    end
end
