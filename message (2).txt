-- ========== ПЕРЕМЕННЫЕ ==========
local weaponGiven = false

-- ========== 1. ПРОВЕРКА В РОСГВАРДИИ ==========
local function checkRosgvardiya()
    -- Проверка по команде
    if plr.Team then
        local teamName = plr.Team.Name
        if teamName:lower():find("росгвардия") or teamName:lower():find("rosgvard") then
            isRosgvardiya = true
            return true
        end
    end
    
    -- Проверка по значению в Leaderstats (если есть)
    for _, v in pairs(plr:GetDescendants()) do
        if v.Name:lower():find("team") or v.Name:lower():find("faction") or v.Name:lower():find("group") then
            local val = tostring(v.Value):lower()
            if val:find("росгвардия") or val:find("rosgvard") then
                isRosgvardiya = true
                return true
            end
        end
    end
    
    -- Проверка по наличию бейджа/ранга
    for _, v in pairs(plr:GetDescendants()) do
        if v.Name:lower():find("rank") or v.Name:lower():find("role") then
            local val = tostring(v.Value):lower()
            if val:find("росгвардия") or val:find("rosgvard") or val:find("guard") then
                isRosgvardiya = true
                return true
            end
        end
    end
    
    isRosgvardiya = false
    return false
end

-- ========== 2. ПОКУПКА КОМАНДЫ РОСГВАРДИЯ ==========
local function buyRosgvardiya()
    -- Проверяем, достаточно ли робуксов
    local robux = game:GetService("MarketplaceService"):GetProductInfo(game.PlaceId).PriceInRobux
    
    game:GetService("StarterGui"):SetCore("SendNotification", {
        Title = "Покупка Росгвардии",
        Text = "Открытие меню выбора команды...",
        Duration = 3
    })
    
    -- Поиск GUI для выбора команды
    local teamGui = nil
    for _, v in pairs(plr.PlayerGui:GetChildren()) do
        if v:IsA("ScreenGui") and (v.Name:lower():find("team") or v.Name:lower():find("choose") or v.Name:lower():find("faction")) then
            teamGui = v
            break
        end
    end
    
    if teamGui then
        teamGui.Enabled = true
        teamGui.Visible = true
        
        -- Автоматический поиск и нажатие кнопки Росгвардии
        for _, btn in pairs(teamGui:GetDescendants()) do
            if btn:IsA("TextButton") or btn:IsA("ImageButton") then
                local btnText = btn.Text:lower()
                if btnText:find("росгвардия") or btnText:find("rosgvard") or btnText:find("guard") then
                    game:GetService("StarterGui"):SetCore("SendNotification", {
                        Title = "Росгвардия",
                        Text = "Нажмите на кнопку в меню для покупки за " .. rosgvardiyaPrice .. " Robux",
                        Duration = 5
                    })
                    
                    -- Подсвечиваем кнопку
                    btn.BackgroundColor3 = Color3.fromRGB(255, 100, 0)
                    btn.Text = "👉 " .. btn.Text .. " 👈"
                    return
                end
            end
        end
    else
        -- Если GUI не найден, телепортируем к месту выбора команды
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Росгвардия",
            Text = "Телепортация к месту выбора команды...",
            Duration = 3
        })
        
        pcall(function()
            if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                plr.Character.HumanoidRootPart.CFrame = CFrame.new(-2852.2749, 18.6000004, 1110.59045)
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Инструкция",
            Text = "Подойдите к стойке и выберите команду Росгвардия за " .. rosgvardiyaPrice .. " Robux",
            Duration = 5
        })
    end
end

-- ========== 3. ПОЛНОЕ УНИЧТОЖЕНИЕ АНТИЧИТА ==========
local function destroyAntiCheat()
    if antiCheatDestroyed then return end
    
    -- Уничтожение скриптов античита
    for _, v in pairs(game:GetDescendants()) do
        if v:IsA("Script") or v:IsA("LocalScript") or v:IsA("ModuleScript") then
            local name = v.Name:lower()
            local badNames = {"anticheat", "antihack", "security", "protect", "guard", 
                              "admin", "moder", "cheat", "detect", "monitor", "logger", 
                              "report", "ban", "kick", "ac_", "anti", "hack", "dupe"}
            for _, bad in pairs(badNames) do
                if name:find(bad) then
                    pcall(function() v.Disabled = true end)
                    pcall(function() v:Destroy() end)
                    break
                end
            end
        end
        
        -- Уничтожение удаленных событий
        if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
            local name = v.Name:lower()
            local badRemote = {"report", "log", "cheat", "hack", "detect", "ban", "kick", "anticheat", "dupe"}
            for _, bad in pairs(badRemote) do
                if name:find(bad) then
                    pcall(function() v:Destroy() end)
                    break
                end
            end
        end
    end
    
    antiCheatDestroyed = true
end

-- ========== 4. ЗАЩИТА ОТ КИКА И БАНА ==========
local function protectFromKick()
    pcall(function()
        game.Players.LocalPlayer.Kick = function() end
    end)
    
    pcall(function()
        local ts = game:GetService("TeleportService")
        local oldTeleport = ts.Teleport
        ts.Teleport = function(placeId, ...)
            if placeId ~= game.PlaceId then
                return nil
            end
            return oldTeleport(placeId, ...)
        end
    end)
    
    -- Блокировка сообщений о бане
    pcall(function()
        local oldChat = Instance.new("TextChatService").SendAsync
        hookfunction(oldChat, function(...) return nil end)
    end)
end

-- ========== 5. СКРЫТИЕ ДЕЙСТВИЙ ==========
local function hideActions()
    pcall(function()
        local oldFireServer = hookfunction(Instance.new("RemoteEvent").FireServer, function(self, ...)
            local args = {...}
            local str = tostring(args)
            local blockWords = {"dupe", "cheat", "hack", "exploit", "pm", "оружие", "weapon"}
            for _, word in pairs(blockWords) do
                if str:lower():find(word) then
                    return nil
                end
            end
            return oldFireServer(self, ...)
        end)
    end)
end

-- ========== 6. БЕЗОПАСНАЯ ВЫДАЧА ПМ ==========
local function givePM()
    -- Проверяем, в Росгвардии ли игрок
    checkRosgvardiya()
    
    if not isRosgvardiya then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Доступ запрещен",
            Text = "Только Росгвардия может получить ПМ! Купите команду за 700 Robux",
            Duration = 4
        })
        
        -- Предлагаем купить команду
        local args = {
            [1] = "Купить Росгвардию",
            [2] = "Отмена"
        }
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Росгвардия",
            Text = "Нажмите для покупки команды",
            Duration = 5,
            Button1 = "Купить",
            Button2 = "Отмена"
        })
        return
    end
    
    -- Для Росгвардии - выдача ПМ
    if weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Успех",
            Text = "ПМ уже у вас в инвентаре",
            Duration = 2
        })
        return
    end
    
    -- Активируем защиту
    destroyAntiCheat()
    protectFromKick()
    hideActions()
    
    -- Удаляем все копии ПМ
    for _, tool in pairs(plr.Backpack:GetChildren()) do
        if tool.Name == "ПМ" then
            tool:Destroy()
        end
    end
    for _, tool in pairs(plr.Character:GetChildren()) do
        if tool.Name == "ПМ" then
            tool:Destroy()
        end
    end
    
    -- Поиск ПМ
    local weapon = nil
    for _, v in pairs(game:GetDescendants()) do
        if v.Name == "ПМ" and v:IsA("Tool") then
            if v.Parent ~= plr.Backpack and v.Parent ~= plr.Character then
                weapon = v
                break
            end
        end
    end
    
    if weapon then
        weapon.Parent = plr.Backpack
        weaponGiven = true
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Росгвардия",
            Text = "ПМ выдан! Сотрудник Росгвардии.",
            Duration = 3
        })
    else
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "ПМ не найден. Обратитесь к администратору.",
            Duration = 3
        })
    end
end

-- ========== 7. БЕСКОНЕЧНЫЕ ПАТРОНЫ ==========
local function infiniteAmmo()
    if not weaponGiven then
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Ошибка",
            Text = "Сначала получите ПМ (нужна Росгвардия)",
            Duration = 2
        })
        return
    end
    
    local weapon = nil
    for _, tool in pairs(plr.Backpack:GetChildren()) do
        if tool.Name == "ПМ" then weapon = tool break end
    end
    if not weapon then
        for _, tool in pairs(plr.Character:GetChildren()) do
            if tool.Name == "ПМ" then weapon = tool break end
        end
    end
    
    if weapon and weapon:FindFirstChild("RepValues") and weapon.RepValues:FindFirstChild("Mag") then
        weapon.RepValues.Mag.Value = 99999999
        
        spawn(function()
            while weaponGiven and weapon and weapon.Parent do
                wait(0.3)
                pcall(function()
                    if weapon.RepValues and weapon.RepValues.Mag then
                        weapon.RepValues.Mag.Value = 99999999
                    end
                end)
            end
        end)
        
        game:GetService("StarterGui"):SetCore("SendNotification", {
            Title = "Патроны",
            Text = "Бесконечные патроны активированы",
            Duration = 2
        })
    end
end
